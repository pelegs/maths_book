\makeatletter
\def\mytransformation{%
	\pgfmathsetmacro{\myX}{\pgf@x}
	\pgfmathsetmacro{\myY}{\pgf@y+0.001*\pgf@x*\pgf@x}
	\setlength{\pgf@x}{\myX pt}
	\setlength{\pgf@y}{\myY pt}
}
\def\testtransform{%
	\pgfmathsetmacro{\myX}{\pgf@x}
	\pgfmathsetmacro{\myY}{\pgf@y+0.00003*\pgf@x*\pgf@x*\pgf@y}
	\setlength{\pgf@x}{\myX pt}
	\setlength{\pgf@y}{\myY pt}
}
\makeatother
